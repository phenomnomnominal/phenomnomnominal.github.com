Clouds = ->
  width = $(window).innerWidth()
  height = $(window).innerHeight()
  
  start_time = new Date().getTime()
  $container = $('.clouds div')

  RENDERER = new THREE.WebGLRenderer(antialias: false)
  RENDERER.setSize width, height
  $container.append RENDERER.domElement
  
  SCENE = new THREE.Scene()

  CAMERA = new THREE.PerspectiveCamera(30, width / height, 1, 5000)
  CAMERA.position.z = 6000  
  SCENE.add CAMERA
  
  texture = THREE.ImageUtils.loadTexture 'images/cloudTexture.png'
  texture.magFilter = THREE.LinearMipMapLinearFilter
  texture.minFilter = THREE.LinearMipMapLinearFilter
  
  fog = new THREE.Fog(0x4584b4, -100, 5000)
  
  material = new THREE.ShaderMaterial(
    uniforms:
      'map':
        type: 't'
        value: texture
        
      'fogColor':
        type: 'c'
        value: fog.color
        
      'fogNear':
        type: 'f'
        value: fog.near
        
      'fogFar':
        type: 'f'
        value: fog.far

    vertexShader: $('#vs').text()
    fragmentShader: $('#fs').text()
    depthTest: false
  )
  
  clouds = new Clouds(32, 8, 128)

  plane = new THREE.Mesh(new THREE.PlaneGeometry(64, 64))
  
  geometry = new THREE.Geometry()
  for i in [0...8000]
    plane.position.x = Math.random() * 1000 - 500
    plane.position.y = -Math.random() * Math.random() * 200 - 15
    plane.position.z = i
    plane.rotation.z = Math.random() * Math.PI
    plane.scale.x = plane.scale.y = Math.random() * Math.random() * 1.5 + 0.5
    THREE.GeometryUtils.merge geometry, plane
    
  mesh = new THREE.Mesh(geometry, material)
  SCENE.add mesh
  
  $(window).resize ->
    width = $(window).innerWidth()
    height = $(window).innerHeight()
    CAMERA.aspect = width / height
    CAMERA.updateProjectionMatrix()
    RENDERER.setSize width, height
    
  $(window).trigger 'resize'
  
  animate = ->
    requestAnimFrame animate
    render()

  render = ->
    position = ( ( new Date().getTime() - start_time ) * 0.03 ) % 4000
    CAMERA.position.z = - position + 8000
    RENDERER.render(SCENE, CAMERA)
  
  animate()

class Clouds
  constructor: (@x, @y, @z, @t = 0) ->
    @size = @x * @y * @z
    @currentGen = hum: '', act: '', cld: ''
    for i in [0...@size]
      @currentGen.hum += if Math.random() > 0.5 then '1' else '0'
      @currentGen.act += if Math.random() > 0.5 then '1' else '0'
      @currentGen.cld += 0
    @clouds =
      current: clouds.call this
    @density =
      current: density.call this, @clouds.current
  
  nextGeneration: ->
    console.time 'growthTime'
    @process growth
    console.timeEnd 'growthTime'
    console.time 'extinctionTime'
    @process extinction
    console.timeEnd 'extinctionTime'
    prevCld = @clouds.current
    prevDen = @density.current
    console.time 'cloudsTime'
    @clouds =
      previous: prevCld
      current: clouds.call this
    console.timeEnd 'cloudsTime'
    console.time 'densityTime'
    @density = 
      previous: prevDen
      current: density.call this, @clouds.current
    console.timeEnd 'densityTime'
    
  logic = 
    and: (str1, str2, res = '') ->
      (res += str1[i] and str2[i]) for i in [0...str1.length]
      res
    or: (str1, str2, res = '') ->
      (res += str1[i] or str2[i]) for i in [0...str1.length]
      res
    not: (str) ->
      str.replace /0|1/g, ((a, m) -> if m is '0' then '1' else '0')
      
  process: (func, newHum = '', newAct = '', newCld = '') ->
    @previousGen = World.Clone @currentGen
    [newHum, newAct, newCld] = func.call(this)
    @currentGen =
      hum: newHum
      act: newAct
      cld: newCld
      
  getCurrentState = ->
    [@previousGen.hum, @previousGen.act, @previousGen.cld]
    
  growth = (nAct = '') ->
    [hum, act, cld] = getCurrentState.call this
    (nAct += "#{if neighboursAct.call(this, i) then 1 else 0}") for i in [0...@size]
    newHum = logic.and hum, logic.not(act)
    newAct = logic.and logic.and(logic.not(act), hum), nAct
    newCld = logic.or hum, act
    [newHum, newAct, newCld]

  neighboursAct = (i) ->
    prev = @previousGen.act
    x = @y * @z
    y = @z
    prev[i + x] || prev[i + y] || prev[i + 1] || prev[i - x] ||
    prev[i - y] || prev[i - 1] || prev[i - 2 * x] || prev[i + 2 * x] ||
    prev[i - 2 * y] || prev[i + 2 * y] || prev[i - 2]

  extinction = ->
    [hum, act, cld] = getCurrentState.call this 
    randHum = @getRandom @size, pHum
    randAct = @getRandom @size, pAct
    randCld = @getRandom @size, 1 - pExt
    newHum = logic.or hum, randHum
    newAct = logic.or act, randAct
    newCld = logic.and cld, randCld
    [newHum, newAct, newCld]
    
  clouds = ->
    cld = @currentGen.cld
    c = []
    for x in [0...@x]
      c[x] ?= []
      for y in [0...@y]
        c[x][y] ?= []
        for z in [0...@z]
          c[x][y][z] = +cld[x * @x + y * @y + z]
    c
  
  density = (clouds) ->
    d = []
    for x in [0...@x]
      d[x] ?= []
      for y in [0...@y]
        d[x][y] ?= []
        for z in [0...@z]
          d[x][y][z] = getDensity clouds, x, y, z
          
  getDensity = (clouds, x, y, z) ->
    weight = (dx, dy, dz) ->
      Math.sqrt((dx * dx) + (dy * dy) + (dz * dz))
    d = 0
    for xs in [x - xSmooth..x + xSmooth]
      for ys in [y - ySmooth..y + ySmooth]
        for zs in [z - zSmooth..z + zSmooth]
          exists = clouds[x + xs]?[y + ys]?[y + zs]?
          d += (if exists and clouds[x + xs][y + ys][z + zs] is 1 then weight(xs, ys, zs) else 0)
    d / ((2 * xSmooth + 1) * (2 * ySmooth + 1) * (2 * zSmooth + 1))

  @Clone: (world) ->
    clone =
      hum: world.hum
      act: world.act
      cld: world.cld
